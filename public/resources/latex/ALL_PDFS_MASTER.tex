\begin{infobox}{What is Full Stack Development?}
Full Stack Development means building \textbf{both the Frontend (what users see)} and the \textbf{Backend (server, database, APIs)} of a web application. A Full Stack Developer handles the complete product from start to finish -- from pixel-perfect UI to database design and server deployment.
\end{infobox}

\section{Phase-by-Phase Learning Roadmap}
% ============================================

\begin{roadmapstep}{1}{Foundation (Months 1-2)}

\textbf{Goal:} Build a solid HTML/CSS/JavaScript foundation before touching any framework.

\vspace{0.3cm}
\textbf{\color{FaranBlue}Month 1 -- HTML \& CSS}
\begin{itemize}[leftmargin=1.5em, itemsep=4pt]
    \item[\ding{62}] HTML5 semantic elements (header, nav, main, section, footer, article)
    \item[\ding{62}] CSS Box Model, Flexbox, CSS Grid
    \item[\ding{62}] Responsive Design with Media Queries
    \item[\ding{62}] CSS Variables, Animations, Transitions
    \item[\ding{62}] \textbf{Project:} Build 3 static websites (Portfolio, Restaurant, Product Page)
\end{itemize}

\vspace{0.3cm}
\textbf{\color{FaranBlue}Month 2 -- JavaScript Fundamentals}
\begin{itemize}[leftmargin=1.5em, itemsep=4pt]
    \item[\ding{62}] Variables, Data Types, Functions, Scope, Closures
    \item[\ding{62}] Arrays, Objects, ES6+ (Arrow Functions, Spread, Destructuring)
    \item[\ding{62}] DOM Manipulation, Events, Forms
    \item[\ding{62}] Promises, Async/Await, Fetch API
    \item[\ding{62}] \textbf{Project:} Todo App, Weather App, Calculator
\end{itemize}

\vspace{0.3cm}
\begin{tcolorbox}[colback=FaranAccent!10, colframe=FaranAccent, boxrule=0.8pt]
\textbf{\color{FaranAccent}Resources:}
\begin{itemize}[leftmargin=1.2em, noitemsep]
    \item The Odin Project (free) -- theodinproject.com
    \item freeCodeCamp.org (free)
    \item JavaScript.info (best JS reference, free)
    \item Traversy Media -- YouTube (free)
\end{itemize}
\end{tcolorbox}

\end{roadmapstep}

\vspace{0.5cm}

\begin{roadmapstep}{2}{React \& Frontend Ecosystem (Months 3-4)}

\textbf{Goal:} Learn React deeply and build dynamic single-page applications.

\vspace{0.3cm}
\textbf{\color{FaranBlue}Month 3 -- React.js Core}
\begin{itemize}[leftmargin=1.5em, itemsep=4pt]
    \item[\ding{62}] JSX, Components (Functional), Props, State
    \item[\ding{62}] useEffect, useState, useContext, useRef, useMemo
    \item[\ding{62}] React Router v6 -- SPA navigation
    \item[\ding{62}] Lifting State, Component Composition patterns
    \item[\ding{62}] \textbf{Project:} Blog App, E-commerce Product Listing, Weather Dashboard
\end{itemize}

\vspace{0.3cm}
\textbf{\color{FaranBlue}Month 4 -- Advanced Frontend}
\begin{itemize}[leftmargin=1.5em, itemsep=4pt]
    \item[\ding{62}] TypeScript with React (interfaces, types, generics)
    \item[\ding{62}] Tailwind CSS utility-first styling
    \item[\ding{62}] State management: Zustand or Redux Toolkit
    \item[\ding{62}] API integration, loading/error states
    \item[\ding{62}] Framer Motion for animations
    \item[\ding{62}] \textbf{Project:} Full Portfolio Website (deploy it!)
\end{itemize}

\end{roadmapstep}

\vspace{0.5cm}

\begin{roadmapstep}{3}{Backend Development (Months 5-6)}

\textbf{Goal:} Build your own REST APIs and connect them to databases.

\vspace{0.3cm}
\textbf{\color{FaranBlue}Month 5 -- Node.js \& Express}
\begin{itemize}[leftmargin=1.5em, itemsep=4pt]
    \item[\ding{62}] Node.js runtime, modules (CommonJS vs ES Modules), npm
    \item[\ding{62}] Express.js -- routing, middleware, error handling
    \item[\ding{62}] REST API design principles (GET, POST, PUT, DELETE)
    \item[\ding{62}] Environment variables with dotenv
    \item[\ding{62}] \textbf{Project:} Build a Notes API with full CRUD
\end{itemize}

\vspace{0.3cm}
\textbf{\color{FaranBlue}Month 6 -- Databases \& Auth}
\begin{itemize}[leftmargin=1.5em, itemsep=4pt]
    \item[\ding{62}] MongoDB \& Mongoose (schemas, models, validation)
    \item[\ding{62}] JWT Authentication (login, register, protected routes)
    \item[\ding{62}] Password hashing with bcrypt
    \item[\ding{62}] File uploads with Multer + Cloudinary
    \item[\ding{62}] \textbf{Project:} Full Auth System (Register/Login/Profile)
\end{itemize}

\end{roadmapstep}

\vspace{0.5cm}

\begin{roadmapstep}{4}{Next.js \& Full Stack (Months 7-8)}

\textbf{Goal:} Combine frontend + backend into production-ready full stack apps.

\vspace{0.3cm}
\begin{itemize}[leftmargin=1.5em, itemsep=4pt]
    \item[\ding{62}] Next.js App Router (layouts, pages, loading, error)
    \item[\ding{62}] Server Components vs Client Components
    \item[\ding{62}] API Routes in Next.js (serverless functions)
    \item[\ding{62}] NextAuth.js for authentication
    \item[\ding{62}] Database integration (MongoDB Atlas / PlanetScale)
    \item[\ding{62}] Deployment on Vercel + Railway
    \item[\ding{62}] SEO, metadata, Open Graph
    \item[\ding{62}] \textbf{Project:} Full Stack SaaS App (with auth, database, payments)

    \item[\ding{62}] Build 3-5 full stack projects for GitHub portfolio
    \item[\ding{62}] Optimize GitHub profile (README, pinned repos, contributions)
    \item[\ding{62}] Create LinkedIn profile with tech keywords
    \item[\ding{62}] Start on Fiverr / Upwork with small projects

\section{UI/UX Fundamentals}

\begin{infobox}{What is UI vs UX?}
\textbf{UI (User Interface)} = The visual layer -- colors, fonts, buttons, spacing, icons. What users \textit{see}.\\[0.3cm]
\textbf{UX (User Experience)} = The feel -- how easy is it to use? How logical is the flow? What users \textit{feel}.\\[0.3cm]
A great product needs \textbf{both}. Beautiful UI with confusing UX = frustrated users. Great UX with ugly UI = users leave.
\end{infobox}

\subsection{The 10 UI/UX Design Principles}
\begin{enumerate}[leftmargin=2em, itemsep=8pt]
    \item \textbf{\color{FaranPurple}Visual Hierarchy} -- Guide user's eye using size, color, weight, and spacing
    \item \textbf{\color{FaranPurple}Consistency} -- Same button styles, fonts, colors throughout. No surprises
    \item \textbf{\color{FaranPurple}Contrast} -- Text must be readable. Minimum 4.5:1 contrast ratio (WCAG AA)
    \item \textbf{\color{FaranPurple}White Space} -- Never fear empty space. It creates focus and breathability
    \item \textbf{\color{FaranPurple}Alignment} -- Every element aligned to a grid. Nothing floating randomly
    \item \textbf{\color{FaranPurple}Proximity} -- Related items grouped together. Unrelated items separated
    \item \textbf{\color{FaranPurple}Feedback} -- Every interaction should have a response (hover, click, loading)
    \item \textbf{\color{FaranPurple}Accessibility} -- Design for everyone including disabled users
    \item \textbf{\color{FaranPurple}Mobile First} -- Design for smallest screen first, then scale up
    \item \textbf{\color{FaranPurple}Less is More} -- Remove everything not absolutely necessary
\end{enumerate}

\section{React Core Concepts}

\subsection{All React Hooks Explained}

\begin{center}
\begin{tabularx}{\textwidth}{|>{\bfseries\color{FaranOrange}}l|X|X|}
\hline
\rowcolor{FaranDark}
\color{white}\textbf{Hook} & \color{white}\textbf{Purpose} & \color{white}\textbf{When to Use} \\
\hline
useState & Store and update local state & Any changing piece of UI data \\
\hline
\rowcolor{FaranLight}
useEffect & Side effects \& lifecycle & API calls, subscriptions, timers \\
\hline
useContext & Access global context data & Theme, auth, language settings \\
\hline
\rowcolor{FaranLight}
useRef & Reference DOM or persist value & Focus input, previous value, timers \\
\hline
useMemo & Memoize expensive computation & Heavy calculations, filtered lists \\
\hline
\rowcolor{FaranLight}
useCallback & Memoize functions & Prevent child re-renders \\
\hline
useReducer & Complex state logic & Form state, multi-step flows \\
\hline
\rowcolor{FaranLight}
useLayoutEffect & DOM measurements & Before paint DOM reads \\
\hline
Custom Hooks & Extract reusable logic & useFetch, useAuth, useLocalStorage \\
\hline
\end{tabularx}

\begin{codebox}{useState Example}
\begin{lstlisting}
// Counter with useState
const [count, setCount] = useState(0);

// Functional update (correct for async)
setCount(prev => prev + 1);

// Object state - always spread!
const [user, setUser] = useState({ name: '', email: '' });
setUser(prev => ({ ...prev, name: 'Faran' }));
\end{lstlisting}
\end{codebox}

\begin{codebox}{useEffect - Common Patterns}
\begin{lstlisting}
// Run once on mount
useEffect(() => {
  fetchData();
}, []);

// Run when dependency changes
useEffect(() => {
  document.title = `Count: ${count}`;
}, [count]);

// Cleanup (subscriptions, timers)
useEffect(() => {
  const timer = setInterval(() => {}, 1000);
  return () => clearInterval(timer); // cleanup
}, []);
\end{lstlisting}
\end{codebox}

\begin{tipbox}
Never put async functions directly in useEffect. Create an inner async function:
\texttt{useEffect(() => \{ const fn = async () => \{...\}; fn(); \}, [])}
\end{tipbox}

\subsection{Server vs Client Components}

\begin{multicols}{2}
\textbf{\color{FaranBlue}Server Component (default)}\\
\begin{itemize}[leftmargin=1em, noitemsep]
    \item Runs on server only
    \item Can fetch data directly
    \item No useState / useEffect
    \item Better performance, smaller bundle
    \item Access server secrets
\end{itemize}
\columnbreak
\textbf{\color{FaranOrange}Client Component ('use client')}\\
\begin{itemize}[leftmargin=1em, noitemsep]
    \item Runs in browser
    \item Can use hooks (state, effects)
    \item Handles user interactions
    \item Access browser APIs
    \item Required for event handlers
\end{itemize}
\end{multicols}

\begin{infobox}{Golden Rule}
Keep components Server by default. Only add \texttt{'use client'} when you \textit{need} browser interactivity. This gives you better performance and SEO automatically.
\end{infobox}

\begin{codebox}{JWT Authentication Middleware}
\begin{lstlisting}
const jwt = require('jsonwebtoken');

const protect = async (req, res, next) => {
  try {
    const token = req.headers.authorization
      ?.split(' ')[1]; // Bearer TOKEN

    if (!token) {
      return res.status(401).json({
        success: false,
        message: 'Not authenticated'
      });
    }

    const decoded = jwt.verify(
      token, process.env.JWT_SECRET
    );
    req.user = decoded;
    next();
  } catch (err) {
    res.status(401).json({ message: 'Invalid token' });
  }
};
\end{lstlisting}
\end{codebox}

\begin{infobox}{API Security Checklist}

    \item[\ding{52}] Implement rate limiting (express-rate-limit)
    \item[\ding{52}] Use helmet.js for security headers
    \item[\ding{52}] CORS properly configured (not wildcard in production)
    \item[\ding{52}] Rotate JWT secrets and use short expiry
    \item[\ding{52}] Never expose stack traces in production responses
    \item[\ding{52}] Use environment variables for all secrets
    \item[\ding{52}] Log errors (winston or pino)
\end{itemize}
