\subsection{URL Naming Conventions}

\begin{infobox}{Good vs Bad API Routes}
\textbf{\color{FaranAccent}\ding{51} Correct REST URLs:}
\begin{itemize}[leftmargin=1.5em, itemsep=2pt]
    \item \texttt{GET /api/users} -- list all users
    \item \texttt{GET /api/users/:id} -- get single user
    \item \texttt{POST /api/users} -- create user
    \item \texttt{PATCH /api/users/:id} -- update user fields
    \item \texttt{DELETE /api/users/:id} -- delete user
    \item \texttt{GET /api/users/:id/posts} -- get user's posts (nested)
\end{itemize}
\vspace{0.3cm}
\textbf{\color{red}\ding{55} Wrong REST URLs:}
\begin{itemize}[leftmargin=1.5em, itemsep=2pt]
    \item \texttt{GET /getUser}, \texttt{POST /deleteUser} -- verbs in URLs are wrong
    \item \texttt{GET /api/User} or \texttt{/api/Users} -- use lowercase, plural nouns
    \item \texttt{GET /api/users/getAll} -- redundant "getAll" is not REST
\end{itemize}
\end{infobox}

// lib/api-response.ts
export function successResponse(data: unknown, message = 'Success', status = 200) {
  return Response.json(
    { success: true, message, data, timestamp: new Date().toISOString() },
    { status }
  );
}

export function errorResponse(message: string, status = 400, errors?: unknown) {
  return Response.json(
    { success: false, message, errors: errors || null, timestamp: new Date().toISOString() },
    { status }
  );
}

// Usage in route handler
// app/api/users/route.ts
export async function GET(request: Request) {
  try {
    const { searchParams } = new URL(request.url);
    const page = Number(searchParams.get('page') ?? 1);
    const limit = Math.min(Number(searchParams.get('limit') ?? 10), 100); // cap at 100

    const [users, total] = await Promise.all([
      prisma.user.findMany({
        skip: (page - 1) * limit,
        take: limit,
        select: { id: true, name: true, email: true, createdAt: true },
        // NEVER return passwords or sensitive fields!
      }),
      prisma.user.count(),
    ]);

    return successResponse({ users, pagination: { page, limit, total,
      totalPages: Math.ceil(total / limit) } });
  } catch (error) {
    console.error('[GET /api/users]', error);
    return errorResponse('Failed to fetch users', 500);
  }
}
\end{lstlisting}

\section{Authentication: JWT + Middleware}

\begin{codebox}{JWT Auth Implementation (Next.js Middleware)}
\begin{lstlisting}
// lib/auth.ts - JWT utilities
import { SignJWT, jwtVerify } from 'jose';

const secret = new TextEncoder().encode(process.env.JWT_SECRET);

export async function signToken(payload: Record<string, unknown>) {
  return await new SignJWT(payload)
    .setProtectedHeader({ alg: 'HS256' })
    .setIssuedAt()
    .setExpirationTime('7d')         // 7 day expiry
    .sign(secret);
}

export async function verifyToken(token: string) {
  const { payload } = await jwtVerify(token, secret);
  return payload;
}

// middleware.ts - protect routes
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';

const PROTECTED_ROUTES = ['/dashboard', '/admin', '/api/protected'];

export async function middleware(request: NextRequest) {
  const isProtected = PROTECTED_ROUTES.some(r => request.nextUrl.pathname.startsWith(r));
  if (!isProtected) return NextResponse.next();

  const token = request.cookies.get('token')?.value
              ?? request.headers.get('Authorization')?.replace('Bearer ', '');

  if (!token) {
    return NextResponse.json({ success: false, message: 'Unauthorized' }, { status: 401 });
  }

  try {
    await verifyToken(token);
    return NextResponse.next();
  } catch {
    return NextResponse.json({ success: false, message: 'Invalid or expired token' }, { status: 401 });
  }
}
\end{lstlisting}
\end{codebox}

\section{Rate Limiting \& Security}

\begin{codebox}{In-Memory Rate Limiter (No Redis Needed)}
\begin{lstlisting}
// lib/rate-limit.ts
const requestMap = new Map<string, { count: number; resetTime: number }>();

export function rateLimit(
  ip: string,
  maxRequests = 10,     // 10 requests
  windowMs = 60_000     // per minute
): { success: boolean; remaining: number } {
  const now = Date.now();
  const entry = requestMap.get(ip);

  if (!entry || now > entry.resetTime) {
    requestMap.set(ip, { count: 1, resetTime: now + windowMs });
    return { success: true, remaining: maxRequests - 1 };
  }

  if (entry.count >= maxRequests) {
    return { success: false, remaining: 0 };
  }

  entry.count++;
  return { success: true, remaining: maxRequests - entry.count };
}

// app/api/contact/route.ts - apply rate limit
export async function POST(request: Request) {
  const ip = request.headers.get('x-forwarded-for') ?? '127.0.0.1';
  const { success, remaining } = rateLimit(ip, 5, 60_000); // 5 per minute

  if (!success) {
    return Response.json(
      { success: false, message: 'Too many requests. Please wait 1 minute.' },
      { status: 429, headers: { 'Retry-After': '60' } }
    );
  }
  // ... handle request
}
\end{lstlisting}
\end{codebox}

\subsection{API Security Checklist}

\begin{rulebox}{Security Checklist -- Every API}
\begin{itemize}[leftmargin=1.5em, itemsep=4pt]
    \item[\ding{51}] \textbf{Validate all inputs} -- never trust user data (use Zod or Joi)
    \item[\ding{51}] \textbf{Sanitize inputs} -- prevent XSS, NoSQL injection
    \item[\ding{51}] \textbf{Rate limit all endpoints} -- especially auth/contact
    \item[\ding{51}] \textbf{Use HTTPS only} -- no plaintext communication
    \item[\ding{51}] \textbf{Set CORS correctly} -- whitelist specific origins
    \item[\ding{51}] \textbf{Never expose stack traces} -- generic error messages
    \item[\ding{51}] \textbf{Set security headers} -- CSP, X-Frame-Options, HSTS
    \item[\ding{51}] \textbf{Hash passwords} -- bcrypt with cost factor 12
    \item[\ding{51}] \textbf{Rotate JWT secrets} -- and invalidate old tokens
    \item[\ding{51}] \textbf{Log all auth events} -- failed logins, signups
\end{itemize}
\end{rulebox}
