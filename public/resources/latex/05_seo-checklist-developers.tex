\section{Structured Data / Schema Markup}

\begin{infobox}{What is Schema Markup?}
Schema.org structured data tells Google \textit{exactly} what your content means. Result: \textbf{Rich Snippets} in search results -- star ratings, FAQs, breadcrumbs, job postings. This dramatically increases click-through rate.
\end{infobox}

\begin{codebox}{Person Schema for Portfolio}
\begin{lstlisting}
// components/seo/PersonSchema.tsx
export function PersonSchema() {
  const schema = {
    "@context": "https://schema.org",
    "@type": "Person",
    "name": "Faran Alam",
    "url": "https://faran-new-portfolio.vercel.app/",
    "jobTitle": "Full Stack Developer",
    "description": "Professional Full Stack Developer...",
    "email": "faran.bsce40@iiu.edu.pk",
    "telephone": "+923334051830",
    "sameAs": [
      "https://github.com/FaranAlam",
      "https://linkedin.com/in/faran-alam",
    ],
    "knowsAbout": ["React", "Next.js", "Node.js", "TypeScript"]
  };

  return (
    <script
      type="application/ld+json"
      dangerouslySetInnerHTML={{ __html: JSON.stringify(schema) }}
    />
  );
}
// Add to app/layout.tsx inside <head>
\end{lstlisting}
\end{codebox}

\section{Core Web Vitals (Performance)}

\begin{center}
\begin{tabularx}{\textwidth}{|>{\bfseries\color{FaranGreen}}l|X|>{\centering\arraybackslash}p{2.5cm}|>{\centering\arraybackslash}p{2.5cm}|}
\hline
\rowcolor{FaranDark}
\color{white}\textbf{Metric} & \color{white}\textbf{What it Measures} & \color{white}\textbf{Good} & \color{white}\textbf{Poor} \\
\hline
LCP & Largest Contentful Paint (loading) & $\leq$ 2.5s & $>$ 4s \\
\hline
\rowcolor{SectionBg}
FID/INP & Input delay / interaction time & $\leq$ 100ms & $>$ 300ms \\
\hline
CLS & Cumulative Layout Shift (stability) & $\leq$ 0.1 & $>$ 0.25 \\
\hline
\end{tabularx}
\end{center}

\subsection{Performance Optimization Checklist}
\begin{itemize}[leftmargin=1.5em, itemsep=6pt]
    \item[\ding{52}] Use Next.js \texttt{<Image>} -- auto WebP, lazy load, proper sizing
    \item[\ding{52}] Always set \texttt{width} and \texttt{height} on images (prevents CLS)
    \item[\ding{52}] Add \texttt{priority} prop to hero/above-fold images (reduces LCP)
    \item[\ding{52}] Use \texttt{next/font} for fonts (eliminates layout shift)
    \item[\ding{52}] Minimize client-side JavaScript (prefer Server Components)
    \item[\ding{52}] Enable Vercel Edge Network caching (automatic on Vercel)
    \item[\ding{52}] Test with PageSpeed Insights: \texttt{pagespeed.web.dev}
    \item[\ding{52}] Use Lighthouse in Chrome DevTools (target 90+ all categories)
\end{itemize}
