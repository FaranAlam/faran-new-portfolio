\section{Sections Every Portfolio Needs}

\begin{sectionbox}{1. Hero Section}
\textbf{Purpose:} First impression. Must answer: Who are you? What do you do? How to contact you?

\vspace{0.3cm}
\textbf{Must Include:}
\begin{itemize}[leftmargin=1.5em, itemsep=3pt]
    \item[\ding{62}] Your name prominently displayed
    \item[\ding{62}] Title/role: ``Full Stack Developer'' or ``React Developer''
    \item[\ding{62}] 1-line value proposition: ``I build fast, beautiful web apps that convert''
    \item[\ding{62}] CTA buttons: ``View My Work'' and ``Hire Me''
    \item[\ding{62}] Professional photo (or high-quality avatar)
    \item[\ding{62}] Social links: GitHub, LinkedIn, Email
    \item[\ding{62}] Animated typing effect for roles
\end{itemize}
\end{sectionbox}

\vspace{0.4cm}

\begin{sectionbox}{2. About Section}
\textbf{Purpose:} Build connection. Let people know the person behind the code.

\vspace{0.3cm}
\textbf{Template for About Text:}\\[0.3cm]
\textit{``I am [name], a [title] from [city]. I specialize in [tech stack] and love building [type of projects]. I started coding [when/why]. When I am not coding, I [hobby/interest]. I am currently [studying/working] and open to [full-time/freelance/remote] opportunities.''}
\end{sectionbox}

\vspace{0.4cm}

\begin{sectionbox}{3. Skills \& Tech Stack}
\textbf{Display as:} Technology icons/logos with progress or category groups. Never boring text lists.

\vspace{0.3cm}
\textbf{Organize by Category:}
\begin{multicols}{3}
\textbf{Frontend:}\\React, Next.js, TypeScript, Tailwind

\textbf{Backend:}\\Node.js, Express, MongoDB

\textbf{Tools:}\\Git, Vercel, Figma, VS Code
\end{multicols}
\end{sectionbox}

\vspace{0.4cm}

\begin{sectionbox}{4. Projects Section (MOST IMPORTANT)}
\textbf{Purpose:} Prove your skills with real work. Quality over quantity.

\vspace{0.3cm}
\textbf{Each Project Card Must Have:}
\begin{itemize}[leftmargin=1.5em, itemsep=3pt]
    \item[\ding{62}] Project screenshot or demo video (auto-play on hover)
    \item[\ding{62}] Project name + 1-line description
    \item[\ding{62}] Tech stack badges (React, Node, MongoDB etc.)
    \item[\ding{62}] \textbf{Live Demo} button (must be working!)
    \item[\ding{62}] \textbf{GitHub} button
    \item[\ding{62}] Key features list (3-5 bullet points)
\end{itemize}

\vspace{0.3cm}
\textbf{Minimum 3 projects. Ideal: 5-6 projects.}
\end{sectionbox}

\vspace{0.4cm}

\begin{sectionbox}{5. Experience / Education}
Timeline format with company/university, role, dates, key achievements. Focus on \textit{what you built} not just job titles.
\end{sectionbox}

\vspace{0.4cm}

\begin{sectionbox}{6. Contact Section}
Working contact form (not just email link) + WhatsApp button + social links. Make it dead simple to reach you.
\end{sectionbox}

\begin{codebox}{Project Setup Commands}
\begin{lstlisting}
# Create Next.js project
npx create-next-app@latest my-portfolio
# Options: TypeScript=Yes, Tailwind=Yes, App Router=Yes

# Install animations
npm install framer-motion

# Install icons
npm install lucide-react react-icons

# Install email service
npm install resend

# Deploy to Vercel
npx vercel
\end{lstlisting}
\end{codebox}

\subsection{File Structure}
\begin{codebox}{Portfolio Folder Structure}

    globals.css        # Global styles
  components/
    sections/          # Hero, About, Projects, etc.
    ui/                # Button, Card, Badge
    layout/            # Navbar, Footer
  public/
    images/            # Project screenshots, avatar
  lib/
    data.ts            # Your projects, skills data
\end{lstlisting}
\end{codebox}
