\section{Choose the Right Database}

\begin{center}
\begin{tabularx}{\textwidth}{|X|X|}
\hline
\rowcolor{FaranDark}
\color{white}\large\textbf{PostgreSQL (Relational)} & \color{white}\large\textbf{MongoDB (Document)} \\
\hline
Tables, rows, columns (strict schema) & Collections, documents, JSON-like (flexible) \\
\hline
\rowcolor{SectionBg}
ACID transactions (guaranteed consistency) & Eventual/configurable consistency \\
\hline
Complex JOINs across tables & Embed related data OR use references \\
\hline
\rowcolor{SectionBg}
Best for: finance, ERP, e-commerce & Best for: CMS, real-time, rapid MVPs \\
\hline
Tools: Prisma ORM, Drizzle ORM & Tools: Mongoose, MongoDB driver \\
\hline
\rowcolor{SectionBg}
Hosted: Supabase, Neon, PlanetScale & Hosted: MongoDB Atlas (free 512MB) \\
\hline
\end{tabularx}
\end{center}

\begin{infobox}{Decision Rule}
\textbf{Use PostgreSQL when:} Data has clear relationships, you need complex reporting, or financial accuracy is critical (payments, inventory).\\[0.2cm]
\textbf{Use MongoDB when:} Building fast with flexible data, working in JavaScript ecosystem, storing user-generated content, or needing horizontal scaling.\\[0.2cm]
\textbf{Use Both:} Many production apps use MongoDB for content + PostgreSQL for transactions (payments, auth).
\end{infobox}

\section{PostgreSQL Schema Design}

\subsection{Normalization Rules}

\begin{center}
\begin{tabularx}{\textwidth}{|>{\bfseries\color{FaranTeal}}l|X|X|}
\hline
\rowcolor{FaranDark}
\color{white}\textbf{Normal Form} & \color{white}\textbf{Requirement} & \color{white}\textbf{Problem it Solves} \\
\hline
1NF & Each column has atomic values, no repeating groups & Removes duplicate columns \\
\hline
\rowcolor{SectionBg}
2NF & 1NF + all non-key columns depend on whole primary key & Removes partial dependencies \\
\hline
3NF & 2NF + no transitive dependencies & Removes indirect dependencies \\
\hline
\rowcolor{SectionBg}
BCNF & 3NF + every determinant is a candidate key & Handles edge cases \\
\hline
\end{tabularx}
\end{center}

\begin{tipbox}
\textbf{Practical rule:} Aim for 3NF in most cases. Over-normalizing creates too many JOINs and hurts performance. Some denormalization (adding redundant columns you query often) is perfectly fine in large-scale apps.
\end{tipbox}

\subsection{Prisma Schema Best Practices}

\begin{codebox}{Well-designed Prisma Schema}
\begin{lstlisting}
model User {
  id        String    @id @default(cuid())  // cuid > autoincrement for distributed
  email     String    @unique
  name      String
  role      Role      @default(USER)        // use enums for fixed values
  posts     Post[]
  profile   Profile?  // optional 1-to-1
  createdAt DateTime  @default(now())
  updatedAt DateTime  @updatedAt            // auto-updated timestamp

  @@index([email])                          // index on frequently queried field
}

model Post {
  id          String   @id @default(cuid())
  title       String
  slug        String   @unique              // for URL-friendly lookups
  content     String
  published   Boolean  @default(false)
  authorId    String
  author      User     @relation(fields: [authorId], references: [id],
                         onDelete: Cascade)  // cascade deletes!
  tags        Tag[]    @relation("PostTags")
  createdAt   DateTime @default(now())

  @@index([authorId])                       // index foreign keys!
  @@index([published, createdAt(sort: Desc)]) // compound index
}

enum Role {
  USER
  ADMIN
  MODERATOR
}
\end{lstlisting}
\end{codebox}

\section{MongoDB Schema Design}

\subsection{Embedding vs Referencing}

\begin{center}
\begin{tabularx}{\textwidth}{|X|X|}
\hline
\rowcolor{FaranDark}
\color{white}\textbf{Embed (denormalize)} & \color{white}\textbf{Reference (normalize)} \\
\hline
Related data inside parent document & Separate collections with ObjectId refs \\
\hline
\rowcolor{SectionBg}
Single query fetches everything fast & Requires populate() / multiple queries \\
\hline
Great for: comments, addresses, tags & Great for: users, posts, large sub-docs \\
\hline
\rowcolor{SectionBg}
Risk: document grows unbounded & Clean separation, easier to update \\
\hline
\end{tabularx}
\end{center}

\begin{codebox}{Mongoose: When to Embed vs Reference}
\begin{lstlisting}
// EMBED: Small, fixed data that belongs to parent
const blogPostSchema = new Schema({
  title: String,
  // Comments EMBEDDED (small, always shown with post)
  comments: [{
    text: String,
    author: String,   // just name, not full user
    createdAt: { type: Date, default: Date.now }
  }],
  // Tags EMBEDDED (simple strings)
  tags: [String],
});

// REFERENCE: Large, independent entities
const blogPostSchema = new Schema({
  title: String,
  // Author REFERENCED (full user object, can be queried alone)
  author: { type: mongoose.Types.ObjectId, ref: 'User', required: true },
  // Category REFERENCED (shared across posts)
  category: { type: mongoose.Types.ObjectId, ref: 'Category' },
});

// Populate references
const post = await Post
  .findById(id)
  .populate('author', 'name email avatar')  // only get needed fields
  .lean();  // returns plain JS object (faster, no mongoose overhead)
\end{lstlisting}
\end{codebox}

\section{Indexing Strategy}

\begin{infobox}{Why Indexing is Critical}
Without indexes, MongoDB/PostgreSQL scans \textbf{every document/row} to find matching data -- a \textbf{collection scan}. With an index, it jumps directly to results in O(log n) time. On a million-record collection, the difference is \textbf{2ms vs 4000ms}.
\end{infobox}

\subsection{MongoDB Indexing}
\begin{codebox}{Essential Index Patterns}
\begin{lstlisting}
// Single field index - most common
userSchema.index({ email: 1 });           // 1=ascending -1=descending
postSchema.index({ createdAt: -1 });      // most recent first

// Compound index (order matters!)
postSchema.index({ author: 1, createdAt: -1 });
// Good for: find posts BY author, sorted by date

// Text index for search
productSchema.index({ name: 'text', description: 'text' });
// Usage: Product.find({ $text: { $search: 'laptop gaming' } })

// Unique index
userSchema.index({ email: 1 }, { unique: true });

// Analyze query performance
await Post.find({ author: id }).explain('executionStats');
// Check: totalDocsExamined should equal nReturned (not full collection)
\end{lstlisting}
\end{codebox}

\subsection{Common Anti-Patterns to Avoid}

\begin{warningbox}
\begin{itemize}[leftmargin=1.5em, itemsep=5pt]
    \item[\ding{55}] \textbf{No indexes on foreign keys} -- every JOIN/populate does full scan
    \item[\ding{55}] \textbf{Storing passwords in plain text} -- always hash with bcrypt (cost 12)
    \item[\ding{55}] \textbf{N+1 query problem} -- fetching 100 posts then 100 separate author queries
    \item[\ding{55}] \textbf{Unbounded arrays in MongoDB} -- arrays growing to 10,000+ elements destroy performance
    \item[\ding{55}] \textbf{SELECT *} -- always select only needed columns/fields
    \item[\ding{55}] \textbf{No pagination} -- returning 50,000 records in one response
    \item[\ding{55}] \textbf{No connection pooling} -- creating new DB connection per request
\end{itemize}
\end{warningbox}
